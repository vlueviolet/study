switch문

값이 정해졌을때 사용하면 됨

var a = 74;
switch (a) {
	case 10 : 
		console.log(’10’);
		break;
	case 74 : 
		console.log(’10’);
		break;
	default : 		//값이 없을 경우
		console.log(’default’);
		break;
}

while문

특정 값을 구할때 사용하면 유용함

var a = 0;
while (a <10) {
	console.log(a);
	a++;
}


for문

for (var a = 0 ; a < $(‘li’).length ; a++) {
	console.log(a);
}

이렇게 하게되면 성능이슈가 발생함, 계속 li를 찾고, 길이를 계산하게됨
var a=0;은 한번만 읽고 더 안읽지만 li길이는 계속 체크하게됨


그래서 성능이슈 해결하기 위한 방법은

for (var a = 0 ; max = $(‘li’).length; a < max ; a++) {	//이렇게 하면 됨
	console.log(a);
}
